git init - инициализирует новый репозиторий Git
git status - показывает состояние рабочего каталога и staging area
git add file - добавляет файлы в staging area для следующего коммита
git commit -m "message" - создает коммит с сообщением
git log - показывает историю коммитов
git branch name - создает новую ветку
git switch name - переключается на указанную ветку
git merge name - объединяет указанную ветку с текущей
git branch -d name - удаляет ветку
git remote add origin url - добавляет удаленный репозиторий
git push -u origin master - отправляет изменения в удаленный репозиторий
git pull - получает изменения из удаленного репозитория
git clone url - клонирует удаленный репозиторий
